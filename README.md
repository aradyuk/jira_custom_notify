# Custom script which works with JIRA API:

### 1. Gem list:
#
> 'pp'

> 'jira-ruby'

> 'dotenv'

> 'json'

> 'colorize'
#
```sh
gem install $gem_name
```

> JIRA_USER=

> JIRA_PASS=

#MAIL:

> MAIL_USER=

> MAIL_PORT=
